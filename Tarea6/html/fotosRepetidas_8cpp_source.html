22 
23  DIR *dir;
24  char parentEdit[1024];

26  strcpy(parentEdit, parent);

28  char *newParent = strcat(parentEdit, subDirName);
29

31 
32  if (dir != NULL) {
33 
34  struct dirent *ent;
35  ent = readdir (dir);
36  while (ent != NULL) {
37 // printf ("%s \n", ent->d_name);
38 
39  if (ent->d_type == DT_REG){ // if entry is a regular file
40  string extension = "jpg";
41  string fname = ent->d_name; // filename
42  // if filename's last characters are extension
43  if (fname.find(extension, (fname.length() - extension.length())) != string::npos){
44 
45  list.push_back(fname); // add filename to results vector
46  }
47  }
48 
49  if (ent->d_type == DT_DIR) {
50  bool punto = strcmp(ent->d_name, ".") == 0;
51  bool puntoPUnto = strcmp(ent->d_name, "..") == 0;
52  if ( !punto && !puntoPUnto ) {
53  getPhotos(newParent, ent->d_name, list);
54  }
55  }
56  ent = readdir (dir);
57  }
58  closedir (dir);
59  }
