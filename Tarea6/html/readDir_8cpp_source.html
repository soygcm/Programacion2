14 /// Establece como máscara de búsqueda \c dir.
15 /// - También establece el valor \c hasNext().
16 void readDir::setMask( const char * dir ) {
17 /* NOTA
18  Cuando la hilera "dir" contiene un slash '/' setMask() la transforma
19  en un '\' que es lo que Windows espera. En este caso, hay que crear
20  una copia de "dir" en donde aparece esa modificación. Como este valor
21  queda almacenado en memoria dinámica, al destruir el objeto o al
22  cambiarle su valor es necesario antes retornar la memoria dinámica
23  a la que "m_pszMask", pues contiene una copia modificada del valor
24  original de "dir".
25 */
26  // "m_bHasSlash" es un sinónimo de "m_bDestroyPtr"
27  bool& m_bHasSlash = m_bDestroyPtr; // nombre local para mejorar la legibilidad
28 
29  if ( m_bHasSlash ) {
30  delete [] ( const_cast<char*>(m_pszMask)); // se vuela el buffer
31  }
32 
33  if ( dir == 0 ) {
34  m_bHasSlash = false;
35  m_bHasNext = false;
36  return;
37  }
38  else if( *dir == 0 ) { // hilera nula
39  m_bHasSlash = false;
40  m_bHasNext = false;
41  return;
42  }
43 
44  if ( 0 == strchr(dir, '/') ) { // el nombre tiene un "slash"
45  m_bHasSlash = false; // el destructor no liberará m_pszMask
46  m_pszMask = dir;
47  }
48  else {
49  m_bHasSlash = true; // hay que usar otra hilera sin '/'
50  unsigned N = strlen( dir );
51  char* dst = new char[N+1];
52  m_pszMask = dst;
53  for ( unsigned i=0; i<=N; ++i ) { // copia la hilera
54  if ( *dir == '/' ) {
55  *dst = '\\'; // corrige el "slash"
56  }
57  else {
58  *dst = *dir; // copia las demás
59  }
60  ++dst; ++dir;
61  }
62  }
63 
64  #if defined( _MSC_VER ) || defined( __GNUC__ )
65  m_handle = _findfirst( m_pszMask , & m_FILE_BLOCK_NEXT );
66  m_bHasNext = ( m_handle != -1L );
67  if ( !m_bHasNext ) { _findclose(m_handle); }
68  #endif
69  #ifdef __BORLANDC__
70  int ALL = WILDCARDS | EXTENSION | FILENAME | DIRECTORY | DRIVE;
71  m_bHasNext = (0 == findfirst( m_pszMask , & m_FILE_BLOCK_NEXT , ALL ) );
72  #endif
73 }

80 void readDir::setMaskWithChop( const char * dir ) {
81  // "m_bHasSlash" es un sinónimo de "m_bDestroyPtr"
82  bool& m_bHasSlash = m_bDestroyPtr; // nombre local para mejorar la legibilidad
83

85  delete [] ( const_cast<char*>(m_pszMask)); // se vuela el buffer
86  }
87  if ( dir == 0 ) {
88  m_bHasSlash = false;
89  m_bHasNext = false;
90  return;
91  }
92  else if( *dir == 0 ) { // hilera nula
93  m_bHasSlash = false;
94  m_bHasNext = false;
95  return;
96  }
97  unsigned N = strlen( dir );
98  unsigned i,numberChopped = N-1; // marca cuántos hay que eliminar del final
99 
100  for ( i=N-1; i>0; --i ) {
101  if ( dir[i]!='\\' && dir[i]!='/' ) {
102  break;
103  }
104  }
105  if ( i==0 && (dir[0]=='\\' || dir[0]=='/') ) {
106  m_bHasSlash = false;
107  m_bHasNext = false;
108  return;
109  }
110 
111  numberChopped = N-(i+1);
112 
113  if ( numberChopped==0 && 0 == strchr(dir, '/') ) {
114  m_bHasSlash = false; // el destructor no liberará m_pszMask
115  m_pszMask = dir;
116  }
117  else {
118  m_bHasSlash = true; // hay que usar otra hilera sin '/'
119  N -= numberChopped;
120  char* dst = new char[N+1];
121  m_pszMask = dst;
122  for ( unsigned i=0; i<N; ++i ) { // copia la hilera
123  if ( *dir == '/' ) {
124  *dst = '\\'; // corrige el "slash"
125  }
126  else {
127  *dst = *dir; // copia las demás
128  }
129  ++dst; ++dir;
130  }
131  *dst = 0; // EOS
132  }
133 
134  #if defined( _MSC_VER ) || defined( __GNUC__ )
135  m_handle = _findfirst( m_pszMask , & m_FILE_BLOCK_NEXT );

148 void readDir::setMask( const std::string& dir ) {
149  unsigned N = dir.length();
150  char* dst = new char[N+1];
151  dir.copy( dst, N,0 );
152  dst[N] = 0; // EOS
153  setMask( dst );
154  if ( m_bDestroyPtr ) { // m_pszMask contiene una copia de "dir"
155  delete [] dst; // deja solo la copia de "m_pszMask"
156  }
157  else {
158  // assert( dst == m_bDestroyPtr );
159  m_bDestroyPtr = true; // se asegura que "dst==m_pszMask" sea destruido
160  }
161  // assert( m_pszMask == dst );
162 }

164 time_t readDir::getTimeWrite() const {
165  #if defined( _MSC_VER ) || defined( __GNUC__ )
166  return m_FILE_BLOCK.time_write;
167  #endif
168  #ifdef __BORLANDC__
169  // | 15.....11 | 10........5 | 4.......0 |
170  // Hours Minutes Seconds/2
171  struct borland_time {
172  unsigned secs_2 : 5; // 0..4 (5 bits)
173  unsigned minutes : 6; // 5..10 (6 bits)
174  unsigned hours : 5; // 1..15 (5 bits)
175  };
176  // | 15..........9 | 8.....5 | 4.......0 |
177  // Years-1980 Month Day
178  struct borland_date {
179  unsigned day : 5; // 0..4 (5 bits)
180  unsigned month : 4; // 5..8 (4 bits)
181  unsigned year_1980 : 7; // 9..15 (7 bits)
182  };
183  struct borland_time * t;
184  struct borland_date * d;
185 
186  t = (borland_time*) ((void*)m_FILE_BLOCK.ff_ftime);
187  d = (borland_date*) ((void*)m_FILE_BLOCK.ff_fdate);
188 
189  struct tm result; // time_t
190  result.tm_sec = t->secs_2 * 2;
191  result.tm_min = t->minutes;
192  result.tm_hour = t->hours;
193  result.tm_mday = -1;
194  result.tm_mon = d->month;
195  result.tm_year = d->year_1980 + 1980 - 1900;
196  result.tm_wday = -1;
197  result.tm_yday = -1;
198  result.tm_isdst = -1;
199  return mktime( & result );

257 /// Retorna el sufijo de \c name que está después del último punto \c '.'.
258 /// - Para \c "dir/ARCHIVO.ext" retorna \c "ext" (sin el punto \c '.').
259 /// - Retorna \c (char*)(0) si el último caracter de \c name es un punto \c '.'.
260 /// - Retorna \c (char*)(0) si no hay extensión.
261 const char* lastExt( const char* name ) {
262  const char *last = name + strlen( name );
263  const char *ext = last;
264  while ( ext != name ) {
265  if ( *ext == '.' ) { ++ext; return ((ext==last) ? 0 : ext); }
266  if ( *ext == '/' || *ext=='\\' ) { return 0; }
267  --ext;
268  }
269  return 0;
270 }
