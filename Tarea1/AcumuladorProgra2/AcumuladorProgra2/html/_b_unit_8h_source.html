945 inline void TestCase::recordFailure(
946  const char* label, const char* fname, int lineno, bool must_copy )
947 {
948  if ( must_copy ) {
949  size_t len = strlen(label)+1;
950  char* str = new char[ len ]; // copies the failure message
951  memcpy( str, label, len );
952  label = str;
953  }
954  m_failureList.push_back( TestCaseFailure ( fname, lineno, label, must_copy ) );
955  m_failure++;
956 /* NOTE: When this method is invoked directly it is because "label" is a
957  constant (char*) string, produced by one of the BUnit_TEST() macro using
958  operator "#" (as in #cond). As this is a constant string, it it not
959  necessary to return its memory when the instance of "TestCase" gets
960  destroyed and, as result of this, field "m_destroy_label" in the
961  corresponding "TestCaseFailure" instance must be false.
962 
963  In contrast, when "label" is generated from std::string, the method
964  recordFailure() to be invoked is not this. Intead the method invoked will be
965  the other version of recordFailure() where parameter "label" is a std::string.
966  That other version always uses "true" for "must_copy", because a (char*) copy
967  of the original std::string is stored in the "TestCaseFailure" instance, and
968  that value should be destroyed when that instance gets destroyed. This fact
969  gets recorded in field "TestCaseFailure::m_destroy_label."
970 
971  This means that no extra work is necessary to handle the two different cases
972  (char*) vs std::string, as the compiler chooses correctly which version of
973  recordFailure() to use.
974 */
975 /* NOTA: Cuando este método es invocado directamente es porque "label" es una
976  hilera constante (char*), producida por alguna de las macro BUnit_TEST()
977  mediante el operador "#" (como en #cond). Como esa hilera es constante, no
978  hace falta retornar su memoria cuando la instancia de "TestCase" sea destruida
979  y, en consecuencia, el campo "m_destroy_label" de la instancia de
980  "TestCaseFailure" correspondiente debe ser "false".
981 
982  En contraposición, si la hilera "label" es generada a partir de std::string, el
983  método recordFailure() que será invocado no es éste. Más bien será la otra
984  versión de recordFailure() la invocada, pues en esa "label" es de tipo
985  std::string. Esa otra versión siempre usa "true" para "must_copy", pues una
986  copia (char*) del valor original std::string queda almacenada en la instancia
987  "TestCaseFailure" y ese valor debe ser destruido cuando esa instancia sea
988  destruida. Este hecho quda registrado en el campo "TestCaseFailure::m_destroy_label."
989 
990  Esto quiere decir que no hace falta hacer trabajo adicional para manejar los
991  dos casos diferentes "char*" vs "std::string", pues el compilador escoge
992  correctamente siempre cuál versión de recordFailure() que hay que usar.
993 */
994 }

1562 template <class TestCase>
1563 bool TestSuite<TestCase>::addTest(TestCase* pT) {
1564  if (pT == 0) {
1565  return false;
1566  }
1567  if ( pT->iAmTestSuite() ) { // ==> addSuite()
1568  addSuite( *( reinterpret_cast< TestSuite<TestCase>* > ( pT ) ) );
1569  return true;
1570  }
1571  else if ( m_allTest.end() != std::find(m_allTest.begin(), m_allTest.end(), pT) ) {
1572  return false; // never trust the client programmer
1573  }
1574  else {
1575  m_allTest.push_back( pT );
