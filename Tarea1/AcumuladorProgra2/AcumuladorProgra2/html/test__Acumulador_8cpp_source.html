45 /// Retorna \c true cuando la suma total de un acumulador es igual a \c total.
46 /// - En caso contrario, simplementer retorna \c false.
47 bool test_Acumulador::Verifica_Suma(long total, unsigned dim, long vec[]) {
48  Acumulador A;

51  A.acumule( vec[i] );
