446 /** Calcula el Máximo Común Divisor de los números \c "x" y \c "y".
447  - <code> mcd(x,y) >= 1 </code> siempre.
448  - MCD <==> GCD: <em> Greatest Common Divisor </em>.
449 
450  \pre
451  <code> (y != 0) </code>
452 
453  \remark
454  Se usa el algoritmo de Euclides para hacer el cálculo.
455 
456  \par Ejemplo:
457  \code
458  2*3*5 == mcd( 2*2*2*2 * 3*3 * 5*5, 2*3*5 )
459  30 == mcd( -3600, -30 )
460  \endcode
461 
462  \dontinclude test_rational.cpp
463  \skipline test::mcd()
464  \until }}
465  \see test_rational<NUM>::test_mcd()
466 */
467 template <class NUM>
468 NUM mcd(NUM x, NUM y) {
469  NUM g = (x < NUM(0) ? -x : x); // trabaja con valores positivos
470  NUM r = (y < NUM(0) ? -y : y); // "r" es el resto
471  NUM temp;
472 
473  do {
474  temp = r;
475  r = g % r;
476  g = temp;
477  } while (NUM(0) != r);
478 
479  return g;
480 }

482 /** Simplifica el numerador y el denomidador.
483  - Transforma el número rational de manera que el numerador y el
484  denominador sean primos relativos, asegurando además que el
485  denominador es siempre positivo.
486  - Si <code>(m_num==0) ==> (m_den==1)</code>.
487  - Simplifica la fracción para que \c m_num y \c m_den sean números
488  primos relativos ie, <code>mcd(m_num,m_den) == 1</code>.
489  - Asegura que \c m_den sea un número positivo.
490  - Restaura la invariante de la clase \c rational.

497 template <class NUM>
498 void rational<NUM>::simplify() {
499  if (num() == NUM(0)) {
500  den() = NUM(1);
501  return;
502  }
503  NUM divisor = mcd(num(), den());
504  if (divisor > NUM(1)) { // ==> (divisor != 0)
505  num() /= divisor;
506  den() /= divisor;
507  }
508  if (den() < NUM(0)) {
509  num() = -num();
510  den() = -den();
511  }
512 }

584 template <class NUM>
585 std::istream& operator>>( std::istream &CIN, rational<NUM>& r ) {
586  char ch; // valor leido, letra por letra, de "CIN"
587  const NUM DIEZ = NUM(10);
588 
589  bool es_positivo = true; // manejo de los signos + y -
590 
591  // se brinca todo hasta el primer dígito
592  do {
593  CIN >> ch;
594  if (ch == '-') { // cambia de signo
595  es_positivo = !es_positivo;
596  }
597  } while (!isdigit(ch));
598 
599  // se traga el numerador
600  r.num() = NUM(0);
601  while (isdigit(ch)) { // convierte a GranNum: izq --> der
602  r.num() = DIEZ * r.num() + (ch-'0');
603  CIN >> ch;
604  }
605 
606  // se brinca los blancos después del numerador
607  while (isspace(ch)) {
608  CIN >> ch;
609  }
610 
611  if (ch ==']') { // es un número entero
612  r.den() = NUM(1);
613  }
614  else {
615  do { // se brinca todo hasta el denominador
616  CIN >> ch;
617  if (ch == '-') {
618  es_positivo = !es_positivo;
619  }
620  } while (!isdigit(ch));
621 
622  // se traga el denominador
623  r.den() = NUM(0);
624  while (isdigit(ch)) {
625  r.den() = DIEZ * r.den() + (ch-'0');
626  CIN >> ch;
627  }
628 
629  // El programa se duerme si en el flujo de entrada
630  // NO aparece el caracter delimitador final "]",
631  // pues la lectura termina hasta encontrar el "]".
632  while (ch != ']') {
