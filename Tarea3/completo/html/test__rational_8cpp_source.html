970 /// Calcula y retorna el residuo \c ( n % m ).
971 template <class INT>
972 inline INT operator% ( const INT& n , const INT& m ) {
973  INT temp = n/m;
974  return INT( n - temp * m );
975 /* Nota:
976  Supongamos que ya tenemos un algoritmo op/(n,m) que calcula la
977  división entera (n/m). A partir de ese algoritmo de división se
978  puede calcular el residuo usando una multiplicación y una resta.
979 
980  El valor que el algoritmo de la división d==(n/m) tiene la
981  siguiente propiedad:
982  - n == d*m + r, en donde 0 <= r < m.
983 
984  A partir de esta ecuación se puede hacerla siguiente deducción:
985  n == d*m + r <==>
986  d*m + r == n <==>
987  r == n - (d*m) <==> d == op/( n , m )
988  r == n - (op/(n,m) * m)
989 
990  O sea que es posible calcular el residuo usando la fórmula
991  (n - (op/(n/m) * m)
992 */
993 }

996 template <class INT>
997 const char * aHilera(const rational<INT>& r) {
998  INT temp = r.num() / r.den();
999  return aHilera(temp);
1000  /* Nota
1001  Como "r" es un número racional, no es posible "sacarle" los
1002  dígitos usando el truco del módulo diez:
1003  digito = numero % 10; // último "digito" de "numero"
1004  Cada vez que se usa la división, la clase rational<> produce un
1005  valor quebrado que nunca llegará a sercero, por lo que algoritmo
1006  aHilera( rational<INT> ) nunca retorna si se le usa con un
1007  argumento racional.
1008  - Por eso hace falta esta especialización de la plantilla.
1009  */

1012 /// Construye una hilera que tiene el valor de \c r en dígitos decimales.
1013 /// - Al terminar de usar la hilera, es responsabilidad del programador
1014 /// usuario retornar la memoria de la hilera retornada.
1015 /// \pre \c INT debe ser una clase entera (no puede ser \c rational<INT>).
1016 template <class INT>
1017 const char * aHilera(const INT& r) {
1018  int signo = 0; // cantidad de campos para almacenar el signo: { 0 , 1 }
1019  if ( r == INT(0) ) {
1020  char * res = new char[2];
1021  res[0] = '0';
1022  res[1] = 0 ;
1023  return res;
1024  }
1025  else if ( r<INT(0) ) {
1026  signo = 1; // el '-' ocupa un campo en la hilera de retorno
1027  }
1028  INT DIEZ = 10, digito;
1029  INT tmp = ( signo==0 ? r : 0-r );
1030  int n=0; // cantidad de dígitos de "r"
1031  while ( tmp != INT(0) ) {
1032  tmp /= DIEZ;
1033  ++n;
1034  }
1035  char * res = new char[n+1+signo];
1036  res[n+signo] = 0;
1037  if (signo == 1) {
1038  res[0] = '-';
1039  ++res; // se "brinca" el signo
1040  tmp = 0-r;
1041  }
1042  else {
1043  tmp = r;
1044  }
1045  while ( n>0 ) {
1046  digito = tmp % DIEZ;
