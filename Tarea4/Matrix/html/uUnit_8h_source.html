30 static // "static" avoids "function already defined" linkage error
31 const char* remove_dir_name( const char* str )
32 {
33  const char *p = str; // p = str+strlen(str);
34  while ( *p != 0 ) { ++p; }
35  while ( *p != '/' && *p != '\\' && *p != ':' ) {
36  if ( p == str ) { return str; }
37  --p;
38  }
