142 /** Verifica la invariante de la clase.
143  - Es posible que la matriz tenga dimensiones nulas, lo que implica que todos los
144  punteros a los vectors paralelos deben ser nulos. Este hecho se marca dándolo
145  el valor \c 0 (cero) al campo \c m_val.
146  - Las matrices quedan almacenadas en un vector de tamaño [M x N].
147  - En todos los algoritmos, "m" o "m_rows" es la cantidad de filas == \c rows()
148  - En todos los algoritmos, "n" o "m_cols" es la cantidad de columnas == \c cols()
149 
150  \par <em>Rep</em> Modelo de la clase
151 \code
152 +---+ / \
153 | 2 | M(i,j) ==> m_val[ (i * m_cols) + j ] | 0 1 2 3 | m_rows == 2
154 +---+ (almacenamiento por filas) | 4 5 6 7 | m_cols == 4
155 | 4 | \ /
156 +---+ +---+---+---+---+---+---+---+---+
157 | *-|-->| 0 | 1 | 2 | 3 | 4 | 5 | 6 | 7 |
158 +---+ +---+---+---+---+---+---+---+---+
159 \endcode
160 
161  \par <em>Rep</em> Modelo de la clase
162 \code
163 +---+
164 | 4 | M(i,j) ==> m_val[ i + (j * m_rows) ] / a e \
165 +---+ (almacenamiento por columnas) | b f | m_rows == 4
166 | 2 | | c g | m_cols == 2
167 +---+ +---+---+---+---+---+---+---+---+ \ d h /
168 | *-|-->| a | b | c | d | e | f | g | h |
169 +---+ +---+---+---+---+---+---+---+---+
170 \endcode
171  - http://www.di-mare.com/adolfo/binder/c03.htm#k1-Rep
172  \remark
173  Libera al programador de implementar el método \c Ok()
174  - http://www.di-mare.com/adolfo/binder/c04.htm#sc11
175 */
176 template<class T>
177 bool check_ok( const Matrix<T>& M ) {
178  if ( M.m_rows == 0 || M.m_cols == 0 ) { // si alguno es cero...
179  if ( M.m_rows != 0 || M.m_cols != 0 ) { // ... los 2 deben serlo
180  return false; /// - Invariante: <code>(M.m_rows == 0) <==> (M.m_cols == 0)</code>
181  }
182  else {
183  if (M.m_val != 0) {
184  return false; /// - Invariante: <code>(M.m_rows == 0) <==> (M.m_val == 0)</code>
185  }
186  return true; // La implementación sí permite (m_val == 0)
187  }
188  }
189 
190  unsigned MxN = M.m_rows * M.m_cols;
191  for (unsigned k=0; k<MxN; ++k) {
192  if ( ! check_ok( M.m_val[k] ) ) {
193  return false; /// - Invariante: <code>check_ok( m_val[k] )</code>
194  }
195  }
196  return true;
197 }

301 /** Copia desde \c "o".
302  - Copia todo el valor de \c "o" sobre \c "*this", de forma que el nuevo valor de
303  \c "*this" sea un duplicado exacto del valor de \c "o"
304  - El valor anterior de \c "*this" se pierde
305  - El objeto \c "o" mantiene su valor anterior
306  - Luego de la copia, cuando el valor de \c "o" cambia, el de \c "*this" no cambiará,
307  y viceversa, pues la copia es una copia profunda; no es superficial
308  - Si \c "*this" es \c "o" entonces su valor no cambia
309  - Después de esta operación siempre ocurre que <code> *this == o </code>
310 
311  \par Complejidad:
312  O( <code> rows() * cols() </code> )
313 
314  \dontinclude test_Matrix.cpp
